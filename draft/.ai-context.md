FLOW:hot_path
  client --GET /{key}--> Middleware --ctx+Timings--> server.ServeHTTP
    --> effectiveReadClient(mode, X-Use-Cache) --> {origin|cache}.Get(key)
      --200--> client(200, Cache-Control:31536000)
      --NoSuchKey/NotFound--> regex_match
        --resize--> originClient.Get(candidate keys) --> resizer.Resize --> putBoth --> client(200)
        --file--> originClient.Get(originalKey) --> putBoth --> client(200)
        --no match--> client(404, max-age=30)
      --other err--> log("cache client error") + regex_match (fail-open)
  Middleware --on inner return--> build Entry --> stdout JSON line
FLOW:write_dual
  putBoth(off) --> s3-put on originClient (= cacheClient) [single]
  putBoth(shadow) --> s3-put-origin then s3-put-cache [both best-effort]
  putBoth(live) --> s3-put-cache then s3-put-origin [both best-effort]
FLOW:fallback_migration
  originClient.Get(primary) --NotFound--> fallback.Get(key)
    --NotFound--> fallback.Get(strip_first_prefix(key))
    --hit--> originClient.Put(primary, key) [lazy migrate, no tags]
    --> caller(data, contentType)
FLOW:worker_trigger
  client --POST /_/worker/trigger {envelope}--> handleWorkerTrigger
    --validate sync--> 400 on first failure
    --validate ok--> 202 Accepted [returns before goroutine]
    --go fn (detached, ctx.Background)--> worker.ProcessBatch
      --for each image--> originClient.Get(image)
        --per-image fail--> skip image
        --ok--> for each size × format --> processOutput
          --resize fail--> skip output
          --ok--> exists-check on destS3Client (= cacheClient in split mode)
            --exists--> skip
            --new--> s3Client.Put AND if distinct destS3Client.Put [per-side log]
FLOW:ERROR
  no_extension --> 404(max-age=30)
  pattern_unmatched --> 404(max-age=30)
  original_not_found --> 404(max-age=30, "Original not found")
  resize_err --> 500(max-age=30, "Resize error")
